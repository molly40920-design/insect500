      return "約 $150 ~ $600";
    }
  };

  return (
    <div className="min-h-screen bg-emerald-50 text-slate-800 font-sans">
      {/* Header */}
      <header className="bg-emerald-700 text-white shadow-md sticky top-0 z-20">
        <div className="max-w-6xl mx-auto px-4 py-4 flex items-center justify-between">
          <div className="flex items-center gap-3">
            <Bug size={32} className="text-emerald-300" />
            <div>
              <h1 className="text-2xl font-bold tracking-wide">昆蟲飼養推薦助手</h1>
            </div>
          </div>
        </div>
      </header>

      <main className="max-w-6xl mx-auto px-4 py-8 grid grid-cols-1 md:grid-cols-12 gap-8">
        
        {/* 左側：篩選條件表單 */}
        <div className="md:col-span-4 space-y-6">
          <div className="bg-white rounded-2xl shadow-sm border border-emerald-100 p-6 sticky top-24">
            <h2 className="text-lg font-bold mb-4 flex items-center gap-2 border-b border-emerald-50 pb-3">
              <Search size={20} className="text-emerald-600" />
              設定您的飼養條件
            </h2>

            {/* 溫度 */}
            <div className="mb-5">
              <label className="block text-sm font-medium text-slate-700 mb-2 flex items-center gap-1">
                <Thermometer size={16} className="text-orange-500" /> 溫度控制能力
              </label>
              <select name="temp" value={criteria.temp} onChange={handleInputChange} className="w-full bg-slate-50 border border-slate-200 text-slate-700 rounded-lg focus:ring-emerald-500 focus:border-emerald-500 p-2.5">
                <option value="all">不限 / 皆可</option>
                <option value="cool">涼爽 / 具備冷氣溫控 (&lt;20°C)</option>
                <option value="room">一般室內常溫 (20-26°C)</option>
                <option value="warm">溫暖 / 夏季無冷氣 (&gt;26°C)</option>
              </select>
            </div>

            {/* 濕度 */}
            <div className="mb-5">
              <label className="block text-sm font-medium text-slate-700 mb-2 flex items-center gap-1">
                <Droplets size={16} className="text-blue-500" /> 環境濕度提供
              </label>
              <select name="humidity" value={criteria.humidity} onChange={handleInputChange} className="w-full bg-slate-50 border border-slate-200 text-slate-700 rounded-lg focus:ring-emerald-500 focus:border-emerald-500 p-2.5">
                <option value="all">不限 / 皆可</option>
                <option value="low">偏乾 / 不易發霉 (&lt;50%)</option>
                <option value="medium">適中 / 偶爾噴水保濕 (50-70%)</option>
                <option value="high">潮濕 / 需定時加濕保水 (&gt;70%)</option>
              </select>
            </div>

            {/* 空間 */}
            <div className="mb-5">
              <label className="block text-sm font-medium text-slate-700 mb-2 flex items-center gap-1">
                <Maximize size={16} className="text-purple-500" /> 可用飼養空間
              </label>
              <select name="space" value={criteria.space} onChange={handleInputChange} className="w-full bg-slate-50 border border-slate-200 text-slate-700 rounded-lg focus:ring-emerald-500 focus:border-emerald-500 p-2.5">
                <option value="all">不限 / 皆可</option>
                <option value="small">桌面小盒 (15cm 以下)</option>
                <option value="medium">中型飼養箱 (15-30cm)</option>
                <option value="large">大型網箱/造景缸 (30cm 以上)</option>
              </select>
            </div>

            {/* 通風 */}
            <div className="mb-5">
              <label className="block text-sm font-medium text-slate-700 mb-2 flex items-center gap-1">
                <Wind size={16} className="text-teal-500" /> 環境通風程度
              </label>
              <select name="ventilation" value={criteria.ventilation} onChange={handleInputChange} className="w-full bg-slate-50 border border-slate-200 text-slate-700 rounded-lg focus:ring-emerald-500 focus:border-emerald-500 p-2.5">
                <option value="all">不限 / 皆可</option>
                <option value="low">一般打孔盒 (保濕優先，適合底棲)</option>
                <option value="high">網箱/大面積通風 (空氣流通優先)</option>
              </select>
            </div>

            {/* 光照 */}
            <div className="mb-5">
              <label className="block text-sm font-medium text-slate-700 mb-2 flex items-center gap-1">
                <Sun size={16} className="text-yellow-500" /> 日照與光線
              </label>
              <select name="light" value={criteria.light} onChange={handleInputChange} className="w-full bg-slate-50 border border-slate-200 text-slate-700 rounded-lg focus:ring-emerald-500 focus:border-emerald-500 p-2.5">
                <option value="all">不限 / 皆可</option>
                <option value="dark">陰暗角落 (幾乎無自然光)</option>
                <option value="indoor">一般室內散光 (無陽光直射)</option>
              </select>
            </div>

            {/* 餌料接受度 */}
            <div className="mb-6">
              <label className="block text-sm font-medium text-slate-700 mb-2 flex items-center gap-1">
                <Utensils size={16} className="text-red-500" /> 餌料準備意願 (重要)
              </label>
              <select name="dietType" value={criteria.dietType} onChange={handleInputChange} className="w-full bg-slate-50 border border-slate-200 text-slate-700 rounded-lg focus:ring-emerald-500 focus:border-emerald-500 p-2.5">
                <option value="all">不限 / 皆可</option>
                <option value="easy">方便現成 (果凍、狗糧、魚飼料)</option>
                <option value="soil">專屬底材 (需買腐植土或發酵木屑)</option>
                <option value="plant">採集植物 (能定期摘新鮮無毒葉片)</option>
                <option value="live">活體餌料 (敢餵食蟑螂、蟋蟀等活餌)</option>
              </select>
            </div>

            {/* ✨ 新增：即時更新開關與按鈕區塊 */}
            <div className="space-y-3">
              <div 
                onClick={toggleAutoSearch}
                className="flex items-center gap-2 cursor-pointer p-2 rounded-lg hover:bg-slate-50 transition-colors select-none"
              >
                <div className={`w-10 h-6 rounded-full p-1 transition-colors duration-300 ease-in-out ${isAutoSearch ? 'bg-emerald-500' : 'bg-slate-300'}`}>
                  <div className={`bg-white w-4 h-4 rounded-full shadow-md transform transition-transform duration-300 ease-in-out ${isAutoSearch ? 'translate-x-4' : 'translate-x-0'}`}></div>
                </div>
                <span className={`text-sm font-medium ${isAutoSearch ? 'text-emerald-700' : 'text-slate-500'}`}>
                  {isAutoSearch ? '⚡ 已開啟即時篩選' : '手動確認後篩選'}
                </span>
              </div>

              <button 
                onClick={handleSearch}
                disabled={isAutoSearch} // 即時模式下禁用按鈕，避免混淆
                className={`w-full font-medium py-3 px-4 rounded-xl transition-all shadow-sm flex justify-center items-center gap-2
                  ${isAutoSearch 
                    ? 'bg-slate-100 text-slate-400 cursor-not-allowed opacity-70' 
                    : 'bg-emerald-600 hover:bg-emerald-700 text-white'
                  }`}
              >
                {isAutoSearch ? (
                  <>
                    <Zap size={18} />
                    結果已即時更新
                  </>
                ) : (
                  <>
                    <CheckCircle2 size={18} />
                    {hasSearched ? "更新篩選條件" : "開始分析推薦"}
                  </>
                )}
              </button>
            </div>
          </div>
        </div>

        {/* 右側：推薦結果展示 (使用 activeCriteria 篩選後的結果) */}
        <div className="md:col-span-8">
          {!hasSearched ? (
            <div className="h-full min-h-[400px] flex flex-col items-center justify-center text-center p-8 bg-white/50 border border-emerald-100/50 rounded-2xl border-dashed">
              <Leaf size={64} className="text-emerald-200 mb-4" />
              <h3 className="text-xl font-bold text-slate-600 mb-2">準備好迎接您的新寵物了嗎？</h3>
              <p className="text-slate-500 max-w-md">
                請在左側設定您能提供的飼養環境條件，點擊「開始分析推薦」，我們將為您配對最適合您的昆蟲夥伴。
              </p>
            </div>
          ) : (
            <div>
              <div className="mb-6 flex justify-between items-end">
                <div>
                  <h2 className="text-2xl font-bold text-emerald-800">推薦結果</h2>
                  <p className="text-slate-600 mt-1">為您找到 {matchedInsects.length} 種適合的昆蟲</p>
                </div>
              </div>

              {matchedInsects.length === 0 ? (
                <div className="bg-white rounded-2xl p-8 text-center shadow-sm border border-red-100">
                  <div className="inline-flex items-center justify-center w-16 h-16 rounded-full bg-red-50 mb-4">
                    <Info size={32} className="text-red-400" />
                  </div>
                  <h3 className="text-lg font-bold text-slate-700 mb-2">沒有找到完全符合的物種</h3>
                  <p className="text-slate-500">
                    您設定的條件可能比較嚴苛，建議您將部分條件設為「不限」再次嘗試！
                  </p>
                </div>
              ) : (
                <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
                  {matchedInsects.map((insect) => (
                    <div key={insect.id} className="bg-white rounded-2xl overflow-hidden shadow-sm hover:shadow-md transition-shadow border border-slate-100 flex flex-col">
                      <div className="p-5 border-b border-slate-50 relative overflow-hidden">
                        <Bug size={100} className="absolute -right-4 -top-4 text-emerald-50 opacity-50" />
                        <div className="flex justify-between items-start relative z-10">
                          <div>
                            <span className="inline-block px-2.5 py-1 bg-emerald-100 text-emerald-800 text-xs font-semibold rounded-full mb-2">
                              {insect.type}
                            </span>
                            <h3 className="text-xl font-bold text-slate-800">{insect.name}</h3>
                            <p className="text-sm italic text-slate-500">{insect.scientificName}</p>
                          </div>
                          <span className={`text-xs font-bold px-2 py-1 rounded border ${
                            insect.difficulty === '新手' ? 'bg-green-50 text-green-700 border-green-200' :
                            insect.difficulty === '進階' ? 'bg-blue-50 text-blue-700 border-blue-200' :
                            'bg-red-50 text-red-700 border-red-200'
                          }`}>
                            {insect.difficulty}
                          </span>
                        </div>
                      </div>
                      
                      <div className="p-5 flex-grow flex flex-col">
                        <p className="text-slate-600 text-sm mb-4 leading-relaxed">
                          {insect.description}
                        </p>
                        
                        <div className="space-y-2 mb-4">
                          <div className="flex items-center text-sm text-slate-600">
                              <Thermometer size={16} className="text-slate-400 mr-2 min-w-4" />
                              <span className="font-medium mr-2">溫度:</span> {insect.tempLabel}
                          </div>
                          <div className="flex items-center text-sm text-slate-600">
                              <Maximize size={16} className="text-slate-400 mr-2 min-w-4" />
                              <span className="font-medium mr-2">空間:</span> {insect.spaceLabel}
                          </div>
                          <div className="flex items-center text-sm text-slate-600">
                            <Utensils size={16} className="text-slate-400 mr-2 min-w-4" />
                            <span className="font-medium mr-2">餌料:</span> {insect.dietLabel}
                          </div>
                        </div>

                        <div className="bg-amber-50 rounded-lg p-3 text-sm mb-4">
                          <span className="font-bold text-amber-800 block mb-1">主要食物：</span>
                          <span className="text-amber-700">{insect.diet}</span>
                        </div>

                        <button 
                          onClick={() => setSelectedInsect(insect)}
                          className="mt-auto w-full py-2 px-3 bg-emerald-50 hover:bg-emerald-100 text-emerald-700 rounded-lg flex items-center justify-center gap-2 font-medium transition-colors text-sm border border-emerald-200"
                        >
                          <BookOpen size={16} />
                          查看飼養指南與市場資訊
                        </button>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>
          )}
        </div>
      </main>

      {/* 飼養指南彈跳視窗 (保持原本邏輯) */}
      {selectedInsect && (
        <div className="fixed inset-0 bg-black/60 z-50 flex items-center justify-center p-4 backdrop-blur-sm" onClick={() => setSelectedInsect(null)}>
          <div 
            className="bg-white rounded-2xl w-full max-w-2xl max-h-[90vh] overflow-y-auto shadow-2xl relative animate-in fade-in zoom-in duration-200" 
            onClick={e => e.stopPropagation()} 
          >
            {/* Modal Header */}
            <div className="bg-emerald-600 p-6 text-white sticky top-0 z-10 flex justify-between items-start">
              <div>
                <h3 className="text-2xl font-bold">{selectedInsect.name}</h3>
                <p className="text-emerald-100 opacity-90">{selectedInsect.scientificName}</p>
              </div>
              <button 
                onClick={() => setSelectedInsect(null)}
                className="p-1 hover:bg-emerald-500 rounded-full transition-colors"
              >
                <X size={24} />
              </button>
            </div>

            {/* Modal Content */}
            <div className="p-6 space-y-8">
              
              <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                {/* 1. 市場資訊區塊 (左側) */}
                <section className="bg-blue-50 border border-blue-100 rounded-xl p-5 h-full">
                  <h4 className="flex items-center gap-2 text-lg font-bold text-blue-800 mb-4 pb-2 border-b border-blue-200">
                    <ShoppingBag size={20} />
                    市場入手情報
                  </h4>
                  <div className="space-y-4">
                    <div>
                      <span className="text-xs font-bold text-blue-500 uppercase tracking-wider block mb-1">💰 參考市價</span>
                      <div className="font-bold text-slate-800 bg-white px-3 py-2 rounded-lg border border-blue-100 inline-block shadow-sm">
                        {getPriceInfo(selectedInsect)}
                      </div>
                    </div>
                    <div>
                      <span className="text-xs font-bold text-blue-500 uppercase tracking-wider block mb-1">入手難易度</span>
                      <div className="font-medium text-slate-700">{getMarketInfo(selectedInsect).level}</div>
                    </div>
                    <div>
                      <span className="text-xs font-bold text-blue-500 uppercase tracking-wider block mb-1">推薦購買管道</span>
                      <div className="text-sm text-slate-600 leading-relaxed">{getMarketInfo(selectedInsect).channel}</div>
                    </div>
                    <div className="text-xs text-blue-400 bg-blue-100/50 p-2 rounded">
                      💡 {getMarketInfo(selectedInsect).desc}
                    </div>
                  </div>
                </section>

                {/* 2. 準備清單區塊 (右側) */}
                <section className="bg-orange-50 border border-orange-100 rounded-xl p-5 h-full">
                  <h4 className="flex items-center gap-2 text-lg font-bold text-orange-800 mb-4 pb-2 border-b border-orange-200">
                    <CheckCircle2 size={20} />
                    飼養前準備清單
                  </h4>
                  <div className="space-y-2">
                    {getChecklist(selectedInsect).map((item, index) => (
                      <div key={index} className="flex items-start gap-2 text-sm">
                        <div className="mt-1 min-w-[12px] w-3 h-3 rounded-sm border border-orange-400"></div>
                        <div>
                          <span className="font-bold text-slate-700 mr-1">{item.item}</span>
                        </div>
                      </div>
                    ))}
                  </div>
                </section>
              </div>

              {/* 3. 詳細飼養指南 (下方) */}
              {!selectedInsect.guide ? (
                <div className="text-center text-slate-400 py-8">
                  <p>抱歉，此物種的詳細飼養教學正在撰寫中...</p>
                </div>
              ) : (
                <div className="space-y-6 pt-4 border-t border-slate-100">
                  <section>
                    <h4 className="flex items-center gap-2 text-lg font-bold text-emerald-800 mb-3 border-b border-emerald-100 pb-2">
                      <Maximize size={20} className="text-emerald-600" />
                      環境佈置 (Setup)
                    </h4>
                    <p className="text-slate-600 leading-relaxed bg-slate-50 p-4 rounded-xl border border-slate-100">
                      {selectedInsect.guide.setup}
                    </p>
                  </section>

                  <section>
                    <h4 className="flex items-center gap-2 text-lg font-bold text-emerald-800 mb-3 border-b border-emerald-100 pb-2">
                      <Leaf size={20} className="text-emerald-600" />
                      日常照顧 (Daily Care)
                    </h4>
                    <p className="text-slate-600 leading-relaxed bg-slate-50 p-4 rounded-xl border border-slate-100">
                      {selectedInsect.guide.care}
                    </p>
                  </section>

                  <section>
                    <h4 className="flex items-center gap-2 text-lg font-bold text-emerald-800 mb-3 border-b border-emerald-100 pb-2">
                      <Star size={20} className="text-emerald-600" />
                      繁殖與備註 (Breeding & Tips)
                    </h4>
                    <p className="text-slate-600 leading-relaxed bg-slate-50 p-4 rounded-xl border border-slate-100">
                      {selectedInsect.guide.breeding}
                    </p>
                  </section>
                </div>
              )}

              <div className="pt-4 flex justify-end">
                <button 
                  onClick={() => setSelectedInsect(null)}
                  className="px-6 py-2 bg-slate-100 hover:bg-slate-200 text-slate-700 font-medium rounded-lg transition-colors"
                >
                  關閉指南
                </button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
